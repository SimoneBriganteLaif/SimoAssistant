jobs:
  # 1) ETL notturno schedulato (sostituisce ETLStack EC2 + add_scheduled_task di prima-power)
  - name: etl-nightly
    kind: scheduled
    command: "run etl --steps LND --steps STG --steps TRN --steps PRS --run-type ALL --send-etl-email"
    schedule: "cron(0 2 * * ? *)"        # EventBridge Scheduler, non Rule
    size: large                           # 2 vCPU / 4 GB su Fargate, ZERO istanze tra i run
    timeout: 3h
    image: etl                            # repo ECR dedicato; tag = digest immutabile (no :latest)
    needs: [db, secret:dwh]               # grant generati: GetSecretValue scoping + SG verso RDS
    enabled: true

  # 2) Job on-demand pesante (sostituisce coci EcsEC2Task usato a mano)
  - name: algo-batch
    kind: on-demand
    command: "run algo"
    size: { cpu: 4096, mem: 16384 }       # taglia esplicita; Fargate (16vCPU/120GB max)
    # compute: ec2                         # OPT-OUT esplicito solo per GPU/oltre-limiti-Fargate
    needs: [db, secret:default]

  # 3) Worker async su coda (sostituisce il background-task nel backend → fuori dal web)
  - name: doc-processor
    kind: queue
    command: "run worker"
    size: medium
    queue: doc-jobs                       # SQS + DLQ creati in automatico
    visibility_timeout: auto              # = 6 × timeout (sano, niente 43200 hardcoded)
    max_receive_count: 5                  # poi → DLQ
    autoscale: false                      # flag: scala sui messaggi in coda (default OFF)
    min_consumers: 0                      # dev può andare a 0; prod default 1

  # 4) Orchestrazione multi-step (sostituisce la factory Step Functions custom)
  - name: full-pipeline
    kind: workflow
    schedule: "cron(0 1 * * ? *)"
    steps:
      - run: etl-nightly                  # riusa il job già definito sopra
      - parallel:
          - run: algo-batch
          - run: doc-processor
      - run: notify                       # un altro job on-demand
